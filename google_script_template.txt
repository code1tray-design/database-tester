/**
 * Simplified Google Apps Script for Student Portal - Essential Data Only
 *
 * Logs only critical indicators:
 * - Login events (timestamp, student, roll, status)
 * - Location verification (timestamp, student, roll, status, distance)
 * - Test completion (timestamp, student, roll, test name, score, time taken, final ranking)
 *
 * SETUP:
 * 1. Create a new Google Sheet
 * 2. Go to Extensions > Apps Script
 * 3. Replace all code with this
 * 4. Deploy as Web App (Execute as: Me, Access: Anyone)
 * 5. Copy the Web App URL to your JavaScript files
 */

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // Handle settings updates
    if (data.type === 'update_settings') {
      var settingsSheet = ss.getSheetByName('Settings');
      if (!settingsSheet) {
        settingsSheet = ss.insertSheet('Settings');
        settingsSheet.appendRow(['Setting', 'Value', 'Last Updated']);
        settingsSheet.getRange(1, 1, 1, 3).setFontWeight('bold').setBackground('#f3f3f3');
      }

      // Update or add GPS setting
      var settingsData = settingsSheet.getDataRange().getValues();
      var gpsRowIndex = -1;

      for (var i = 1; i < settingsData.length; i++) {
        if (settingsData[i][0] === 'gps_enabled') {
          gpsRowIndex = i + 1; // +1 because array is 0-indexed but sheet is 1-indexed
          break;
        }
      }

      if (gpsRowIndex === -1) {
        // Add new setting
        settingsSheet.appendRow(['gps_enabled', data.enabled, new Date()]);
      } else {
        // Update existing setting
        settingsSheet.getRange(gpsRowIndex, 2).setValue(data.enabled);
        settingsSheet.getRange(gpsRowIndex, 3).setValue(new Date());
      }

      return ContentService
        .createTextOutput(JSON.stringify({ 'result': 'success', 'message': 'Settings updated' }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // Create main data sheet if it doesn't exist
    var sheet = ss.getSheetByName('StudentData');
    if (!sheet) {
      sheet = ss.insertSheet('StudentData');
      sheet.appendRow(['Timestamp', 'Type', 'Student Name', 'Roll Number', 'Test Name', 'Score', 'Time Taken', 'Final Ranking', 'Location Status', 'Distance (km)', 'Status']);
      sheet.getRange(1, 1, 1, 11).setFontWeight('bold').setBackground('#f3f3f3');
    }

    var timestamp = new Date();

    // Handle different data types with essential fields only
    if (data.type === 'login') {
      sheet.appendRow([
        timestamp,
        'Login',
        data.studentName,
        data.rollNumber,
        '',
        '',
        '',
        '',
        '',
        '',
        data.statusText || 'Success'
      ]);
    } else if (data.type === 'location_log') {
      sheet.appendRow([
        timestamp,
        'Location',
        data.studentName,
        data.rollNumber,
        '',
        '',
        '',
        '',
        data.status,
        data.distanceKm,
        'Verified'
      ]);
    } else if (data.type === 'test') {
      // Extract essential test data only
      var testName = data.testName || 'Unknown Test';
      var score = data.score || 0;
      var timeTaken = data.duration || '';
      var finalRanking = data.percentage || '';

      // For simulator, include key metrics in ranking
      if (testName.includes('Simulator')) {
        finalRanking = `${data.percentage} (Profit: ₹${data.profit || 0}, Market: ${data.marketShare || 0}%)`;
      }

      sheet.appendRow([
        timestamp,
        'Test Result',
        data.studentName,
        data.rollNumber,
        testName,
        score,
        timeTaken,
        finalRanking,
        '',
        '',
        'Completed'
      ]);
    }

    return ContentService
      .createTextOutput(JSON.stringify({ 'result': 'success' }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    // Log errors to a separate sheet for debugging
    var errorSheet = ss.getSheetByName('Errors');
    if (!errorSheet) {
      errorSheet = ss.insertSheet('Errors');
      errorSheet.appendRow(['Timestamp', 'Error', 'Data']);
    }
    errorSheet.appendRow([new Date(), error.toString(), e.postData ? e.postData.contents : 'No data']);

    return ContentService
      .createTextOutput(JSON.stringify({ 'result': 'error', 'error': error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // Handle GPS settings update
    if (e.parameter.action === 'updateGps') {
      var enabled = e.parameter.enabled === 'true';
      var settingsSheet = ss.getSheetByName('Settings');
      if (!settingsSheet) {
        settingsSheet = ss.insertSheet('Settings');
        settingsSheet.appendRow(['Setting', 'Value', 'Last Updated']);
        settingsSheet.getRange(1, 1, 1, 3).setFontWeight('bold').setBackground('#f3f3f3');
      }

      // Update or add GPS setting
      var settingsData = settingsSheet.getDataRange().getValues();
      var gpsRowIndex = -1;

      for (var i = 1; i < settingsData.length; i++) {
        if (settingsData[i][0] === 'gps_enabled') {
          gpsRowIndex = i + 1; // +1 because array is 0-indexed but sheet is 1-indexed
          break;
        }
      }

      if (gpsRowIndex === -1) {
        // Add new setting
        settingsSheet.appendRow(['gps_enabled', enabled, new Date()]);
      } else {
        // Update existing setting
        settingsSheet.getRange(gpsRowIndex, 2).setValue(enabled);
        settingsSheet.getRange(gpsRowIndex, 3).setValue(new Date());
      }

      return ContentService
        .createTextOutput(JSON.stringify({ 'success': true, 'message': 'GPS setting updated' }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    var sheet = ss.getSheetByName('StudentData');

    if (!sheet) {
      // Create the sheet if it doesn't exist and return empty data
      sheet = ss.insertSheet('StudentData');
      sheet.appendRow(['Timestamp', 'Type', 'Student Name', 'Roll Number', 'Test Name', 'Score', 'Time Taken', 'Final Ranking', 'Location Status', 'Distance (km)', 'Status']);
      sheet.getRange(1, 1, 1, 11).setFontWeight('bold').setBackground('#f3f3f3');

      // Return empty data structure
      return ContentService
        .createTextOutput(JSON.stringify({
          'result': 'success',
          'data': [],
          'logins': [],
          'locations': [],
          'tests': [],
          'summary': {
            'total_logins': 0,
            'total_tests': 0,
            'total_locations': 0,
            'unique_students': 0
          }
        }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // Get all data for admin dashboard
    var data = sheet.getDataRange().getValues();
    var headers = data[0];
    var rows = data.slice(1);

    // Convert to objects for easier processing
    var records = rows.map(function(row) {
      var obj = {};
      headers.forEach(function(header, i) {
        obj[header.toLowerCase().replace(/\s+/g, '')] = row[i];
      });
      return obj;
    });

    // Separate data by type for admin panel
    var loginRecords = records.filter(r => r.type === 'Login');
    var locationRecords = records.filter(r => r.type === 'Location');
    var testRecords = records.filter(r => r.type === 'Test Result');

    // Get GPS settings
    var gpsEnabled = true; // Default to enabled
    var settingsSheet = ss.getSheetByName('Settings');
    if (settingsSheet) {
      var settingsData = settingsSheet.getDataRange().getValues();
      for (var i = 1; i < settingsData.length; i++) {
        if (settingsData[i][0] === 'gps_enabled') {
          gpsEnabled = settingsData[i][1] === true || settingsData[i][1] === 'true';
          break;
        }
      }
    }

    return ContentService
      .createTextOutput(JSON.stringify({
        'result': 'success',
        'data': records,
        'logins': loginRecords,
        'locations': locationRecords,
        'tests': testRecords,
        'settings': {
          'gps_enabled': gpsEnabled
        },
        'summary': {
          'total_logins': loginRecords.length,
          'total_tests': testRecords.length,
          'total_locations': locationRecords.length,
          'unique_students': [...new Set(records.map(r => r.rollnumber))].length
        }
      }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({ 'result': 'error', 'error': error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
